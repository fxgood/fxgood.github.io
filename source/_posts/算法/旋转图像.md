[leetcode48 旋转图像](https://leetcode.cn/problems/rotate-image/)

```java
class Solution {
    int[][] matrix;
    public void rotate(int[][] matrix) {
        this.matrix=matrix;
        flipByMainDiagonal();        
        flipRight();
    }

    void swap(int x1,int y1,int x2,int y2){
        int t=matrix[x1][y1];
        matrix[x1][y1]=matrix[x2][y2];
        matrix[x2][y2]=t;
    }

    public void flipRight(){
        int n=matrix.length;
        for(int i=0;i<n;i++)
            for(int j=0;j<n/2;j++){
                swap(i,j,i,n-1-j);
            }
    }

    public void flipByMainDiagonal(){
        int n=matrix.length;
        for(int i=0;i<n;i++)
            for(int j=0;j<i;j++){
                swap(i,j,j,i);
            }
    }

}
```
