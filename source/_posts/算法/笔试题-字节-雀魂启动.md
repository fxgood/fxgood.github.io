1. 卡牌类的题目,用数组来处理是最好的(直接用数组作为哈希表)
2. 仔细审题! 这一题里边说了每张卡牌最多4张,就因为这个问题没看到,导致我找了半个小时的bug

```java
import java.util.*;
public class Main{
    public static void main(String[]args){
        new Main().solve();
        for(Integer a:ans)
            System.out.print(a+" ");
        if(ans.size()==0)
            System.out.print(0);
    }
    int[]card=new int[10];
    static List<Integer>ans=new ArrayList<>();
    void solve(){
        Scanner sc=new Scanner(System.in);
        while(sc.hasNext()){
            card[sc.nextInt()]++;
        }
        for(int k=1;k<=9;k++){
            if(card[k]==4)
                continue;
            card[k]++;
            for(int i=1;i<=9;i++){
                if(card[i]>=2){
                    card[i]-=2;
                    boolean flag=check(0);
                    card[i]+=2;
                    if(flag){
                        ans.add(k);
                        break;
                    }
                }
            }
            card[k]--;
        }

    }
    //对于每个数字来说,不是刻子就是顺子
    boolean check(int k){
        if(k==10)    //说明1~9都是0
            return true;
        if(card[k]==0)
            return check(k+1);
        //当成顺子
        boolean res=false;
        if(k<=7 && card[k+1]>0 && card[k+2]>0){
            card[k]--;
            card[k+1]--;
            card[k+2]--;
            res=check(k);
            card[k]++;
            card[k+1]++;
            card[k+2]++;
        }
        if(res)
            return true;
        //当成刻子
        if(card[k]>=3){
            card[k]-=3;
            res=check(k);
            card[k]+=3;
        }
        return res;
    }

}
```
